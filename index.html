    // ----------------- 1. 高级流体动力学 -----------------

    const canvas = document.getElementById('canvas-stage');

    const ctx = canvas.getContext('2d');

    let atoms = [];


    function initAtoms() {

        canvas.width = window.innerWidth;

        canvas.height = window.innerHeight;

        atoms = [];

        for(let i=0; i<15; i++) {

            atoms.push({

                x: Math.random() * canvas.width,

                y: Math.random() * canvas.height,

                vx: (Math.random()-0.5) * 0.8,

                vy: (Math.random()-0.5) * 0.8,

                r: Math.random() * 300 + 200,

                color: i % 2 === 0 ? '#0033ff' : '#7000ff'

            });

        }

    }


    function animateAtoms() {

        ctx.fillStyle = '#000';

        ctx.fillRect(0,0,canvas.width, canvas.height);

        ctx.globalCompositeOperation = 'screen';

        atoms.forEach(a => {

            a.x += a.vx; a.y += a.vy;

            if(a.x < -a.r) a.x = canvas.width + a.r;

            if(a.x > canvas.width + a.r) a.x = -a.r;

            if(a.y < -a.r) a.y = canvas.height + a.r;

            if(a.y > canvas.height + a.r) a.y = -a.r;

            

            let g = ctx.createRadialGradient(a.x, a.y, 0, a.x, a.y, a.r);

            g.addColorStop(0, a.color);

            g.addColorStop(1, 'transparent');

            ctx.fillStyle = g;

            ctx.beginPath();

            ctx.arc(a.x, a.y, a.r, 0, Math.PI*2);

            ctx.fill();

        });

        requestAnimationFrame(animateAtoms);

    }

    window.onresize = initAtoms;
